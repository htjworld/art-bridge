FROM node:22-alpine AS builder
WORKDIR /app

COPY package*.json ./
COPY tsconfig.json ./
COPY index.ts ./
COPY lib.ts ./

RUN npm install
RUN npm run build

FROM node:22-alpine AS release
WORKDIR /app

# dist 폴더 전체를 /app으로 복사
COPY --from=builder /app/dist/ ./
COPY --from=builder /app/package.json ./
COPY --from=builder /app/package-lock.json ./

ENV NODE_ENV=production
RUN npm ci --omit=dev

# 파일 구조 확인
RUN ls -la

# /app/index.js 실행 (dist/ 없이)
CMD ["node", "index.js"]
